14#[cfg(test)]
15fn base64_encode(input: &[u8]) -> String {
16    const ALPHA: &[u8] = b"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
17    let mut out = String::with_capacity((input.len() + 2) / 3 * 4);
18    input.chunks(3).for_each(|chunk| {
19        let b0 = u32::from(chunk[0]);
20        let b1 = if chunk.len() > 1 {
21            u32::from(chunk[1])
22        } else {
23            0
24        };
25        let b2 = if chunk.len() > 2 {
26            u32::from(chunk[2])
27        } else {
28            0
29        };
30        let n = (b0 << 16) | (b1 << 8) | b2;
31        out.push(char::from(ALPHA[((n >> 18) & 0x3f) as usize]));
32        out.push(char::from(ALPHA[((n >> 12) & 0x3f) as usize]));
33        out.push(if chunk.len() > 1 {
34            char::from(ALPHA[((n >> 6) & 0x3f) as usize])
35        } else {
36            '='
37        });
38        out.push(if chunk.len() > 2 {
39            char::from(ALPHA[(n & 0x3f) as usize])
40        } else {
41            '='
42        });
43    });
44    out
45}

91    /// Returns `true` when the STS credentials expire within 5 minutes.
92    /// Satisfies T-STR-04.
93    pub fn creds_expiring_soon(&self) -> bool {
94        match parse_iso8601_secs(&self.credentials.expires_at) {
95            Some(expires_at_secs) => {
96                let now = std::time::SystemTime::now()
97                    .duration_since(std::time::UNIX_EPOCH)
98                    .unwrap_or_default()
99                    .as_secs();
100                expires_at_secs.saturating_sub(now) < 300
101            }
102            None => {
103                // Unparseable expires_at: treat as already expired to be safe.
104                true
105            }
106        }
107    }
108}

110/// Parse a subset of ISO 8601 UTC timestamps into Unix seconds.
111/// Handles the common AWS format `"YYYY-MM-DDTHH:MM:SSZ"`.
112fn parse_iso8601_secs(s: &str) -> Option<u64> {
113    // Expected: "2026-04-01T12:00:00Z" (20 chars, no sub-second, UTC only)
114    let s = s.trim_end_matches('Z');
115    let s = s.trim_end_matches("+00:00");
116    let nums: Vec<u64> = s
117        .split(|c: char| !c.is_ascii_digit())
118        .filter(|p| !p.is_empty())
119        .map(|p| p.parse().ok())
120        .collect::<Option<Vec<_>>>()?;
121    if nums.len() < 6 {
122        return None;
123    }
124    let (y, mo, d, h, mi, sec) = (nums[0], nums[1], nums[2], nums[3], nums[4], nums[5]);
125    // Approximate days since epoch (good enough for expires_at comparison).
126    let days = days_since_epoch(y, mo, d)?;
127    Some(days * 86400 + h * 3600 + mi * 60 + sec)
128}

130/// Format a Unix timestamp as an ISO 8601 UTC string (`"YYYY-MM-DDTHH:MM:SSZ"`).
131/// Accurate for 1970-2199.
132fn unix_secs_to_iso8601(secs: u64) -> String {
133    let tod = secs % 86400;
134    let mut days = secs / 86400;
135    let hh = tod / 3600;
136    let mm = (tod % 3600) / 60;
137    let ss = tod % 60;
138
139    let mut year = 1970u64;
140    loop {
141        let is_leap = (year % 4 == 0 && year % 100 != 0) || year % 400 == 0;
142        let yd = if is_leap { 366u64 } else { 365u64 };
143        if days < yd {
144            break;
145        }
146        days -= yd;
147        year += 1;
148    }
149    let is_leap = (year % 4 == 0 && year % 100 != 0) || year % 400 == 0;
150    const MDAYS: [u64; 12] = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
151    let mut month = 1u64;
152    loop {
153        let dim = if month == 2 && is_leap {
154            29u64
155        } else {
156            MDAYS[(month - 1) as usize]
157        };
158        if days < dim {
159            break;
160        }
161        days -= dim;
162        month += 1;
163    }
164    let day = days + 1;
165    format!("{year:04}-{month:02}-{day:02}T{hh:02}:{mm:02}:{ss:02}Z")
166}

168fn now_iso8601() -> String {
169    let secs = std::time::SystemTime::now()
170        .duration_since(std::time::UNIX_EPOCH)
171        .unwrap_or_default()
172        .as_secs();
173    unix_secs_to_iso8601(secs)
174}

176/// Days from 1970-01-01 to the given date (Gregorian, valid 1970–2099).
177fn days_since_epoch(y: u64, mo: u64, d: u64) -> Option<u64> {
178    if mo < 1 || mo > 12 || d < 1 || d > 31 || y < 1970 {
179        return None;
180    }
181    // Days per month (non-leap); February adjusted below.
182    const DAYS: [u64; 12] = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
183    let is_leap = (y % 4 == 0 && y % 100 != 0) || y % 400 == 0;
184    let year_days: u64 = (1970..y)
185        .map(|yr| {
186            if (yr % 4 == 0 && yr % 100 != 0) || yr % 400 == 0 {
187                366
188            } else {
189                365
190            }
191        })
192        .sum();
193    let month_days: u64 = (1..mo)
194        .map(|m| {
195            if m == 2 && is_leap {
196                29
197            } else {
198                DAYS[(m - 1) as usize]
199            }
200        })
201        .sum();
202    Some(year_days + month_days + d - 1)
203}

301/// POST to `/runs` to register a new run with the Sentinel API.
302///
303/// On failure the caller should log a warning and disable streaming;
304/// local stdout output continues normally (Section 11 error handling).
305pub fn start_run(
306    agent: &ureq::Agent,
307    api_base: &str,
308    token: &str,
309    metadata: &JobMetadata,
310    pid: Option<i32>,
311    host: &HostInfo,
312    cloud: &CloudInfo,
313) -> Result<RunContext, String> {
314    // The API `command` field is `string | null` ("JSON array encoded in TEXT").
315    // Serialize the Vec<String> to a JSON string; send None when no command was given.
316    let command_json: Option<String> = if metadata.command.is_empty() {
317        None
318    } else {
319        serde_json::to_string(&metadata.command).ok()
320    };
321
322    // `pid` is tracked locally but is not part of the `RunCreate` API schema.
323    let _ = pid;
324
325    let payload = StartRunRequest {
326        metadata: MetadataPayload {
327            job_name: metadata.job_name.as_deref(),
328            project_name: metadata.project_name.as_deref(),
329            stage_name: metadata.stage_name.as_deref(),
330            task_name: metadata.task_name.as_deref(),
331            team: metadata.team.as_deref(),
332            env: metadata.env.as_deref(),
333            language: metadata.language.as_deref(),
334            orchestrator: metadata.orchestrator.as_deref(),
335            executor: metadata.executor.as_deref(),
336            external_run_id: metadata.external_run_id.as_deref(),
337            container_image: metadata.container_image.as_deref(),
338            tags: &metadata.tags,
339            command: command_json,
340        },
341        host,
342        cloud,
343    };
344
345    let url = format!("{api_base}/runs");
346    let body = serde_json::to_string(&payload)
347        .map_err(|e| format!("failed to serialize start_run payload: {e}"))?;
348
349    let mut response = agent
350        .post(&url)
351        .header("Authorization", &format!("Bearer {token}"))
352        .header("Content-Type", "application/json")
353        .send(body.as_bytes())
354        .map_err(|e| format!("start_run POST failed: {e}"))?;
355
356    let text = response
357        .body_mut()
358        .read_to_string()
359        .map_err(|e| format!("start_run read body failed: {e}"))?;
360
361    let resp: StartRunResponse = serde_json::from_str(&text).map_err(|e| {
362        format!(
363            "start_run parse response failed: {e} ({} bytes)",
364            text.len()
365        )
366    })?;
367
368    Ok(RunContext {
369        run_id: resp.run_id,
370        upload_uri_prefix: resp.upload_uri_prefix,
371        credentials: UploadCredentials {
372            access_key_id: resp.upload_credentials.access_key,
373            secret_access_key: resp.upload_credentials.secret_key,
374            session_token: resp.upload_credentials.session_token,
375            expires_at: resp
376                .upload_credentials
377                .expiration
378                .unwrap_or_else(|| "2099-01-01T00:00:00Z".to_string()),
379        },
380    })
381}

383/// POST to `/runs/{run_id}/credentials/refresh` to obtain fresh STS credentials.
384///
385/// Updates `ctx.credentials` in place on success.
386pub fn refresh_credentials(
387    agent: &ureq::Agent,
388    api_base: &str,
389    token: &str,
390    ctx: &mut RunContext,
391) -> Result<(), String> {
392    let url = format!("{api_base}/runs/{}/refresh-credentials", ctx.run_id);
393    let mut response = agent
394        .post(&url)
395        .header("Authorization", &format!("Bearer {token}"))
396        .send(b"" as &[u8])
397        .map_err(|e| format!("credential refresh POST failed: {e}"))?;
398
399    let text = response
400        .body_mut()
401        .read_to_string()
402        .map_err(|e| format!("credential refresh read body failed: {e}"))?;
403
404    let resp: RefreshCredentialsResponse = serde_json::from_str(&text).map_err(|e| {
405        format!(
406            "credential refresh parse failed: {e} ({} bytes)",
407            text.len()
408        )
409    })?;
410
411    ctx.credentials = UploadCredentials {
412        access_key_id: resp.upload_credentials.access_key,
413        secret_access_key: resp.upload_credentials.secret_key,
414        session_token: resp.upload_credentials.session_token,
415        expires_at: resp
416            .upload_credentials
417            .expiration
418            .unwrap_or_else(|| "2099-01-01T00:00:00Z".to_string()),
419    };
420    Ok(())
421}

423/// POST to `/runs/{run_id}/finish` to mark the run complete.
424///
425/// Dispatches to the correct schema variant based on whether S3 uploads occurred:
426///
427/// - `uploaded_uris` non-empty → `RunFinishS3` (`data_source="s3"`, `data_uris=[...]`).
428///   The `BatchUploader` already flushed the final remaining samples to S3 on
429///   shutdown before this function is called, so `uploaded_uris` contains every batch.
430///   `remaining_csv` is ignored in this path.
431///
432/// - `uploaded_uris` empty → `RunFinishInline` (`data_source="inline"`, `data_csv=...`).
433///   Used for short runs or when all S3 uploads failed.  `remaining_csv` is the
434///   raw CSV of all collected samples (never base64-encoded).
435pub fn close_run(
436    agent: &ureq::Agent,
437    api_base: &str,
438    token: &str,
439    ctx: &RunContext,
440    exit_code: Option<i32>,
441    remaining_csv: Option<String>,
442    uploaded_uris: &[String],
443) -> Result<(), String> {
444    // "finished" for success/clean exit (including SIGTERM), "failed" for non-zero.
445    let run_status = match exit_code {
446        Some(0) | None => "finished",
447        Some(_) => "failed",
448    };
449    let finished_at = Some(now_iso8601());
450
451    let url = format!("{api_base}/runs/{}/finish", ctx.run_id);
452    let body = if uploaded_uris.is_empty() {
453        // Inline route: no S3 batches uploaded.
454        // data_csv must be the raw CSV string (not base64) per the API schema.
455        let payload = CloseRunInlineRequest {
456            exit_code,
457            run_status,
458            finished_at,
459            data_source: "inline",
460            data_csv: remaining_csv.unwrap_or_default(),
461        };
462        serde_json::to_string(&payload)
463            .map_err(|e| format!("failed to serialize close_run inline payload: {e}"))?
464    } else {
465        // S3 route: all batches (including final flush) are in uploaded_uris.
466        let payload = CloseRunS3Request {
467            exit_code,
468            run_status,
469            finished_at,
470            data_source: "s3",
471            data_uris: uploaded_uris.to_vec(),
472        };
473        serde_json::to_string(&payload)
474            .map_err(|e| format!("failed to serialize close_run s3 payload: {e}"))?
475    };
476
477    // Send plain JSON -- no body-level gzip.  The Sentinel API (FastAPI) does
478    // not decompress Content-Encoding: gzip on incoming request bodies, so
479    // sending compressed bytes causes a 422.  This matches the Python reference
480    // which calls requests.post(url, json=payload) with no Content-Encoding.
481    let response = agent
482        .post(&url)
483        .header("Authorization", &format!("Bearer {token}"))
484        .header("Content-Type", "application/json")
485        .send(body.as_bytes())
486        .map_err(|e| format!("close_run POST failed: {e}"))?;
487
488    let status = response.status();
489    if status != 200 {
490        return Err(format!("close_run received HTTP {status}: expected 200"));
491    }
492
493    Ok(())
494}

879    // Helper: spin up a mock TCP server that mimics the real Sentinel /finish endpoint:
880    // reads the full request, validates presence of required fields, returns a proper
881    // RunFinishResponse JSON on success (200) or a 422 JSON error if required fields
882    // are missing.  Returns the parsed request body JSON for assertions.
883    fn capture_close_run_body(exit_code: Option<i32>, csv: Option<&str>) -> String {
884        use std::io::{Read as _, Write as _};
885        use std::sync::mpsc;
886        use std::time::Duration;
887
888        let listener = std::net::TcpListener::bind("127.0.0.1:0").unwrap();
889        let port = listener.local_addr().unwrap().port();
890        let (tx, rx) = mpsc::channel::<Vec<u8>>();
891
892        std::thread::spawn(move || {
893            if let Ok((mut stream, _)) = listener.accept() {
894                let mut buf = Vec::<u8>::new();
895                let mut tmp = [0u8; 4096];
896                loop {
897                    let n = stream.read(&mut tmp).unwrap_or(0);
898                    if n == 0 {
899                        break;
900                    }
901                    buf.extend_from_slice(&tmp[..n]);
902                    if let Some(sep) = buf.windows(4).position(|w| w == b"\r\n\r\n") {
903                        let header_str = String::from_utf8_lossy(&buf[..sep]).to_ascii_lowercase();
904                        let cl = header_str
905                            .lines()
906                            .find_map(|l| {
907                                l.trim()
908                                    .strip_prefix("content-length:")
909                                    .and_then(|v| v.trim().parse::<usize>().ok())
910                            })
911                            .unwrap_or(0);
912                        if buf.len() >= sep + 4 + cl {
913                            break;
914                        }
915                    }
916                }
917                // Parse the body and decide: 200 with RunFinishResponse, or 422.
918                let body_start = buf
919                    .windows(4)
920                    .position(|w| w == b"\r\n\r\n")
921                    .map(|p| p + 4)
922                    .unwrap_or(buf.len());
923                let body_str = String::from_utf8_lossy(&buf[body_start..]);
924                let parsed: serde_json::Value =
925                    serde_json::from_str(&body_str).unwrap_or(serde_json::Value::Null);
926                // Real API requires data_source and data_csv for RunFinishInline.
927                let valid = parsed.get("data_source").and_then(|v| v.as_str()) == Some("inline")
928                    && parsed.get("data_csv").is_some();
929                let (status_line, resp_body) = if valid {
930                    let run_status = parsed
931                        .get("run_status")
932                        .and_then(|v| v.as_str())
933                        .unwrap_or("finished");
934                    let ec = parsed
935                        .get("exit_code")
936                        .and_then(|v| v.as_i64())
937                        .map(|v| v as i32);
938                    (
939                        "HTTP/1.1 200 OK",
940                        finish_response_json("run-spec-test", run_status, ec),
941                    )
942                } else {
943                    (
944                        "HTTP/1.1 422 Unprocessable Entity",
945                        r#"{"detail":[{"loc":["body"],"msg":"field required","type":"value_error.missing"}]}"#.to_string(),
946                    )
947                };
948                tx.send(buf).ok();
949                let http = format!(
950                    "{status_line}\r\nContent-Type: application/json\r\nContent-Length: {}\r\n\r\n{}",
951                    resp_body.len(),
952                    resp_body
953                );
954                stream.write_all(http.as_bytes()).ok();
955            }
956        });
957
958        let agent = ureq::config::Config::builder()
959            .timeout_global(Some(Duration::from_secs(30)))
960            .build()
961            .new_agent();
962        let ctx = RunContext {
963            run_id: "run-spec-test".to_string(),
964            upload_uri_prefix: "s3://b/p".to_string(),
965            credentials: UploadCredentials {
966                access_key_id: "k".to_string(),
967                secret_access_key: "s".to_string(),
968                session_token: "t".to_string(),
969                expires_at: "2099-01-01T00:00:00Z".to_string(),
970            },
971        };
972        let _ = close_run(
973            &agent,
974            &format!("http://127.0.0.1:{port}"),
975            "token",
976            &ctx,
977            exit_code,
978            csv.map(String::from),
979            &[], // no S3 uploads → inline route
980        );
981        let raw = rx.recv().expect("mock server did not capture request");
982        // Extract the JSON body (everything after \r\n\r\n).
983        let body_start = raw
984            .windows(4)
985            .position(|w| w == b"\r\n\r\n")
986            .map(|p| p + 4)
987            .unwrap_or(0);
988        String::from_utf8_lossy(&raw[body_start..]).to_string()
989    }

1186    // Helper: call close_run with S3 URIs, return the captured request body JSON.
1187    fn capture_close_run_s3_body(exit_code: Option<i32>, uris: &[&str]) -> String {
1188        use std::io::{Read as _, Write as _};
1189        use std::sync::mpsc;
1190        use std::time::Duration;
1191
1192        let listener = std::net::TcpListener::bind("127.0.0.1:0").unwrap();
1193        let port = listener.local_addr().unwrap().port();
1194        let (tx, rx) = mpsc::channel::<Vec<u8>>();
1195
1196        std::thread::spawn(move || {
1197            if let Ok((mut stream, _)) = listener.accept() {
1198                let mut buf = Vec::<u8>::new();
1199                let mut tmp = [0u8; 4096];
1200                loop {
1201                    let n = stream.read(&mut tmp).unwrap_or(0);
1202                    if n == 0 {
1203                        break;
1204                    }
1205                    buf.extend_from_slice(&tmp[..n]);
1206                    if let Some(sep) = buf.windows(4).position(|w| w == b"\r\n\r\n") {
1207                        let hdr = String::from_utf8_lossy(&buf[..sep]).to_ascii_lowercase();
1208                        let cl = hdr
1209                            .lines()
1210                            .find_map(|l| {
1211                                l.trim()
1212                                    .strip_prefix("content-length:")
1213                                    .and_then(|v| v.trim().parse::<usize>().ok())
1214                            })
1215                            .unwrap_or(0);
1216                        if buf.len() >= sep + 4 + cl {
1217                            break;
1218                        }
1219                    }
1220                }
1221                // Real API: 200 RunFinishResponse for a valid S3 payload.
1222                let body_start = buf
1223                    .windows(4)
1224                    .position(|w| w == b"\r\n\r\n")
1225                    .map(|p| p + 4)
1226                    .unwrap_or(buf.len());
1227                let body_str = String::from_utf8_lossy(&buf[body_start..]);
1228                let parsed: serde_json::Value =
1229                    serde_json::from_str(&body_str).unwrap_or(serde_json::Value::Null);
1230                let valid = parsed.get("data_source").and_then(|v| v.as_str()) == Some("s3")
1231                    && parsed.get("data_uris").is_some();
1232                let run_status = parsed
1233                    .get("run_status")
1234                    .and_then(|v| v.as_str())
1235                    .unwrap_or("finished");
1236                let ec = parsed
1237                    .get("exit_code")
1238                    .and_then(|v| v.as_i64())
1239                    .map(|v| v as i32);
1240                let (status_line, resp_body) = if valid {
1241                    (
1242                        "HTTP/1.1 200 OK",
1243                        finish_response_json("run-s3-test", run_status, ec),
1244                    )
1245                } else {
1246                    (
1247                        "HTTP/1.1 422 Unprocessable Entity",
1248                        r#"{"detail":[{"msg":"field required","type":"value_error.missing"}]}"#
1249                            .to_string(),
1250                    )
1251                };
1252                tx.send(buf).ok();
1253                let http = format!(
1254                    "{status_line}\r\nContent-Type: application/json\r\nContent-Length: {}\r\n\r\n{}",
1255                    resp_body.len(),
1256                    resp_body
1257                );
1258                stream.write_all(http.as_bytes()).ok();
1259            }
1260        });
1261
1262        let agent = ureq::config::Config::builder()
1263            .timeout_global(Some(Duration::from_secs(30)))
1264            .build()
1265            .new_agent();

1267            run_id: "run-s3-test".to_string(),
1268            upload_uri_prefix: "s3://b/p".to_string(),
1269            credentials: UploadCredentials {
1270                access_key_id: "k".to_string(),
1271                secret_access_key: "s".to_string(),
1272                session_token: "t".to_string(),
1273                expires_at: "2099-01-01T00:00:00Z".to_string(),
1274            },
1275        };
1276        let uploaded: Vec<String> = uris.iter().map(|s| (*s).to_string()).collect();
1277        let _ = close_run(
1278            &agent,
1279            &format!("http://127.0.0.1:{port}"),
1280            "token",
1281            &ctx,
1282            exit_code,
1283            None, // remaining_csv unused in S3 route
1284            &uploaded,
1285        );
1286        let raw = rx.recv().expect("mock server did not capture request");
1287        let body_start = raw
1288            .windows(4)
1289            .position(|w| w == b"\r\n\r\n")
1290            .map(|p| p + 4)
1291            .unwrap_or(0);
1292        String::from_utf8_lossy(&raw[body_start..]).to_string()
